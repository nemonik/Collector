  def start
    IO.popen("java -jar ../../OOoConversionSrvc/executable/OOoConversionSrvc-1.0-SNAPSHOT-executable.jar 2>&1 &") unless running?

    Timeout::timeout(60) do
      until (running? && listening?)
        sleep 0.5
      end
      return true
    end
  rescue Timeout::Error
    return false
  end
