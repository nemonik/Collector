  def listening?
    begin
      client_socket = TCPSocket.new(HOSTNAME, PORT)
      client_socket.close
      LOG.debug('OOoConversionSrvc is listening.')
      return true
    rescue Exception => e
      LOG.debug("#{e.class} : #{e.message}")
    end

    LOG.debug('OOoConversionSrvc is not listening.')
    return false
  end
