    def status_of_workers
      values = Array.new
      #@mutex.synchronize {
      @workers.each {|w|

        value = ''
        value += "#{w.get_name} is "
        if w.busy?
          value += 'busy'
          value = value.red
        else
          value += 'idle'
          value = value.green
        end
        values.push(value)
      }

      #}

      values
    end
