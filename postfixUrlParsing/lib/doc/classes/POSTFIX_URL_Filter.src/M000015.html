  def process_compressed(file_name, compression)

    @log.debug("processing #{compression} compressed #{file_name}...")

    extracted_folder = File.join(File.dirname(file_name), Guid.new.to_s)
    Dir.new(extracted_folder)
    FileUtils.cp(file_name, extracted_folder)

    tmp_file_name = File.join(extracted_folder,File.basename(file_name))

    if (compression == 'zip')
      unzip(tmp_file_name)
    elsif (compression == 'gzip')
      unzip(tmp_file_name)
    elsif (compression == 'bzip2')
      unbzip2(tmp_file_name)
    elsif (compression == 'tar')
      untar(tmp_file_name)
    else
      @log.error("#{compression} is an unhandled form of compression")
    end

    FileUtils.rm(tmp_file_name)

    Dir.foreach(extracted_folder) { |file_name|
      file_name = File.join(extracted_folder, filename)

      process_file(file_name) if file_name.file?
    }

  end
